255 template< unsigned COMPONENT_COUNT, unsigned ...COMPONENT_BITS >
256 struct Packed_Layout
257 {
258 private:
259
260 template< unsigned TARGET_INDEX >
261 static constexpr unsigned value_at(unsigned index) { return 0; }
262
263 template< unsigned TARGET_INDEX, typename ...TYPE_PACK >
264 static constexpr unsigned value_at(unsigned index, unsigned value, const TYPE_PACK & ...value_pack)
265 {
266 return index == TARGET_INDEX ? value : value_at< TARGET_INDEX >(index + 1, value_pack...);
267 }
268
269 static constexpr unsigned sum() { return 0; }
270
271 template< typename ...TYPE_PACK >
272 static constexpr unsigned sum(unsigned value, const TYPE_PACK & ...value_pack)
273 {
274 return value + sum(value_pack...);
275 }
276
277 template< unsigned START_INDEX >
278 static constexpr unsigned sum_beyond(unsigned index) { return 0; }
279
280 template< unsigned START_INDEX, typename ...TYPE_PACK >
281 static constexpr unsigned sum_beyond(unsigned index, unsigned value, const TYPE_PACK & ...value_pack)
282 {
283 return (index > START_INDEX ? value : 0) + sum_beyond< START_INDEX >(index + 1, value_pack...);
284 }
285
286 public:
287
288 static constexpr unsigned component_count = COMPONENT_COUNT;
289 static constexpr unsigned bits = sum(COMPONENT_BITS...);
290
291 using Component_Type = void;
292 using Composite_Type = typename Composite_Type< bits >::Type;
293
294 template< unsigned COMPONENT_INDEX >
295 struct Component_Traits
296 {
297 static constexpr unsigned index = COMPONENT_INDEX;
298 static constexpr unsigned bits = value_at < COMPONENT_INDEX >(0, COMPONENT_BITS...);
299 static constexpr unsigned shift = sum_beyond< COMPONENT_INDEX >(0, COMPONENT_BITS...);
300 static constexpr Composite_Type mask = (1 << bits) - 1;
301 static constexpr float maxf = float((1 << bits) - 1);
302 };

304 };

519 Additive_Primaries& set

521 const float& normalized_r,
522 const float& normalized_g,
523 const float& normalized_b
524 )
525 {
526 value =
527 (Composite_Type(normalized_r * red_traits::maxf) << red_traits::shift) |
528 (Composite_Type(normalized_g * green_traits::maxf) << green_traits::shift) |

535 Composite_Type green() const { return (value >> green_traits::shift) & green_traits::mask; }
536 Composite_Type blue() const { return (value >> blue_traits::shift) & blue_traits::mask; }
