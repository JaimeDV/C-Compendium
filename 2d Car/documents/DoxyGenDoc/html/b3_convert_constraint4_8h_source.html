8void b3PlaneSpace1(b3Float4ConstArg n, b3Float4* p, b3Float4* q)
9{
10 if (b3Fabs(n.z) > 0.70710678f)
11 {
12 // choose p in y-z plane
13 float a = n.y * n.y + n.z * n.z;
14 float k = 1.f / sqrt(a);
15 p[0].x = 0;
16 p[0].y = -n.z * k;
17 p[0].z = n.y * k;
18 // set q = n x p
19 q[0].x = a * k;
20 q[0].y = -n.x * p[0].z;
21 q[0].z = n.x * p[0].y;
22 }
23 else
24 {
25 // choose p in x-y plane
26 float a = n.x * n.x + n.y * n.y;
27 float k = 1.f / sqrt(a);
28 p[0].x = -n.y * k;
29 p[0].y = n.x * k;
30 p[0].z = 0;
31 // set q = n x p
32 q[0].x = -n.z * p[0].y;
33 q[0].y = n.z * p[0].x;
34 q[0].z = a * k;
35 }
36}

51float calcJacCoeff(b3Float4ConstArg linear0, b3Float4ConstArg linear1, b3Float4ConstArg angular0, b3Float4ConstArg angular1,
52 float invMass0, const b3Mat3x3* invInertia0, float invMass1, const b3Mat3x3* invInertia1)
53{

62void setConstraint4(b3Float4ConstArg posA, b3Float4ConstArg linVelA, b3Float4ConstArg angVelA, float invMassA, b3Mat3x3ConstArg invInertiaA,
63 b3Float4ConstArg posB, b3Float4ConstArg linVelB, b3Float4ConstArg angVelB, float invMassB, b3Mat3x3ConstArg invInertiaB,

139 if (i < src->m_worldNormalOnB.w)

141 dstC->m_worldPos[i] = src->m_worldPosB[i];
